#   macOS:
#     brew install portaudio
#
#   Windows:
#     Download wheel dari https://www.lfd.uci.edu/~gohlke/pythonlibs/#pyaudio
#     pip install PyAudio‑0.2.14‑cpXX‑cpXX‑win_amd64.whl
#
# Setelah system library terinstall, jalankan:
#   pip install -r requirements-audio.txt
